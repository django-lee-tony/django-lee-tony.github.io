svg正确使用方式

svg-sprite-loader 下载

npm i svg-sprite-loader 或者 yarn add svg-sprite-loader

新建vue.config.js 创建src/assets/svg 文件夹 将svg放入该文件夹中

const path = require("path");
function resolve(dir) {
    return path.join(__dirname, ".", dir);
}
module.exports={
    ...
    chainwebpack:config =>{
        ...
    config.module.rules.delete("svg"); //重点:删除默认配置中处理svg,
            //const svgRule = config.module.rule('svg')
            //svgRule.uses.clear()
            config.module
                .rule("svg-sprite-loader")
                .test(/\.svg$/)
                .include.add(resolve("src/assets/svg")) //处理svg目录
                .end()
                .use("svg-sprite-loader")
                .loader("svg-sprite-loader")
                .options({
                    symbolId: "icon-[name]"
                });
    }
}

封装组件

新建index.vue

//index.vue
<template>
    <svg :class="svgClass" aria-hidden="true">
        <use :xlink:href="iconName"/>
    </svg>
</template>
<script>
    export default {
        name: 'SvgIcon',
        props: {
            iconClass: { //接收svg名称 注意就是svg文件名
                type: String,
                required: true
            },
            className: { //svg图标的属性 如通过font-sizee属性控制icon大小
                type: String,
                default: ''
            }
        },
        computed: {
            iconName() {
                return `#icon-${this.iconClass}`
            },
            svgClass() {
                if (this.className) {
                    return 'svg-icon ' + this.className
                } else {
                    return 'svg-icon'
                }
            }
        }
    }
</script>
<style scoped>
    .svg-icon {
        width: 1em;
        height: 1em;
        vertical-align: -0.15em;
        fill: currentColor;
        overflow: hidden;
    }
</style>

index.js文件

//使用webpack提供的 require.context方法 引入所有的svg
import svgIcon from "./index.vue"
const requireAll = requireContext => requireContext.keys().map(requireContext)
const req = require.context('../../assets/svg', false, /\.svg$/)
requireAll(req)
// 暴露方法
export default {
    install(Vue){
        Vue.component('svgIcon',svgIcon);
    }
}
